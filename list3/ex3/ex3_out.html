def generate_small_world(N, avg_degree, p, dataset):
    k = int(avg_degree/2)
    GWS = nx.watts_strogatz_graph(N, k, p, seed=None)
    dataset.append(GWS)

def degree_distribution(G):
    vk = dict(G.degree())
    vk = list(vk.values()) # we get only the degree values
    maxk = np.max(vk)
    kvalues= np.arange(0,maxk+1) # possible values of k
    Pk = np.zeros(maxk+1) # P(k)
    for k in vk:
        Pk[k] = Pk[k] + 1
    Pk = Pk/sum(Pk) # the sum of the elements of P(k) must to be equal to one
    return kvalues,Pk

def second_momment_of_degree_distribution(G):
    k,Pk = degree_distribution(G)
    M = sum((k**2)*Pk)
    return M

def verify_small_world(N, k):
    vcc = []
    vl = []
    vq = []
    for q in np.arange(0, 1, 0.02):
        G = nx.watts_strogatz_graph(N, k, q, seed=None)
        cc = nx.average_clustering(G)
        vcc.append(cc)
        l = nx.average_shortest_path_length(G)
        vl.append(l)
        vq.append(q)
    plt.figure()
    plt.xscale('log')
    plt.plot(vq,np.array(vcc)/vcc[0],'b', label='Clustering coefficient')
    plt.plot(vq,np.array(vl)/vl[0],'r', label='Average vertex-vertex distance')
    plt.xlabel("rewring probability p", fontsize=20)
    plt.ylabel("$cc/cc_0$, $\ell/\ell_0$", fontsize=20)
    plt.title("Small-world networks", fontsize=20)
    plt.legend()
    plt.grid(True)
    plt.show()

def plot_2nd_momment_of_degree_distribution(N, k):
    vq = []
    vm = []
    for q in np.arange(0, 1, 0.02):
        G = nx.watts_strogatz_graph(N, k, q, seed=None)
        m = second_momment_of_degree_distribution(G)
        vm.append(m)
        vq.append(q)
    plt.figure()
    plt.xscale('log')
    plt.plot(vq,np.array(vm)/vm[0],'b')
    plt.xlabel("Rewiring probability p", fontsize=20)
    plt.ylabel("m", fontsize=20)
    plt.title("Second Momment of Degree Distribution", fontsize=20)
    plt.grid(True)
    plt.show()

def plot_degree_distribution(N, k, q, n_networks, title):
    vk = list()
    for i in range(0, n_networks):
        G = nx.watts_strogatz_graph(N, k, q, seed=None)       
        vi = dict(G.degree())
        vi = list(vi.values()) # we get only the degree values
        vk.append(vi)
    maxk = np.max(vk)
    ks= np.arange(0,maxk+1) # possible values of k
    Pk = np.zeros(maxk+1) # P(k)
    for k in vk:
        Pk[k] = Pk[k] + 1
    Pk = Pk/sum(Pk) # the sum of the elements of P(k) must to be equal to one
    plt.figure()
    plt.plot(ks,Pk,'bo')
    plt.xlabel("k", fontsize=20)
    plt.ylabel("P(k)", fontsize=20)
    plt.title(title, fontsize=20)
    plt.grid(True)
    plt.show(True)

def plot_avg_length_scaling(Nmax, k, q, title):
    vd = []
    vn = []
    Nmax = 1000
    for n in np.arange(10, Nmax, Nmax/20):
        G = nx.watts_strogatz_graph(N, k, q, seed=None)  
        vd.append(nx.average_shortest_path_length(G))
        vn.append(n)
    plt.figure()
    plt.loglog(vn,vd,'-or',basex=10,basey=10, label='SW')
    x = np.arange(1, Nmax, Nmax/20)
    plt.loglog(x,np.log10(x),'--b', label='ER', basex=10,basey=10)
    plt.xlabel("N", fontsize=20)
    plt.ylabel("d", fontsize=20)
    plt.title(title, fontsize=20)
    plt.grid(True)
    plt.legend()
    plt.show()
