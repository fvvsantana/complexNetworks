def second_momment_of_degree_distribution(G):
    k,Pk = degree_distribution(G)
    M = sum((k**2)*Pk)
    return M

def degree_distribution(G):
    vk = dict(G.degree())
    vk = list(vk.values()) # we get only the degree values
    maxk = np.max(vk)
    kvalues= np.arange(0,maxk+1) # possible values of k
    Pk = np.zeros(maxk+1) # P(k)
    for k in vk:
        Pk[k] = Pk[k] + 1
    Pk = Pk/sum(Pk) # the sum of the elements of P(k) must to be equal to one
    return kvalues,Pk

#Degree distribution for multiple networks
def degree_distribution_list(G, n_networks):
    vk = list()
    for i in range(0, n_networks):
        vi = dict(G[i].degree())
        vi = list(vi.values()) # we get only the degree values
        vk.append(vi)
    maxk = np.max(vk)
    kvalues= np.arange(0,maxk+1) # possible values of k
    Pk = np.zeros(maxk+1) # P(k)
    for k in vk:
        Pk[k] = Pk[k] + 1
    Pk = Pk/sum(Pk) # the sum of the elements of P(k) must to be equal to one
    return kvalues,Pk

def plot_degree_distribution(ks, Pk, p):
    plt.figure()
    plt.plot(ks,Pk,'bo', label='Data')
    plt.xlabel("k", fontsize=20)
    plt.ylabel("P(k)", fontsize=20)
    plt.title("Degree distribution", fontsize=20)
    plt.grid(True)

    pk = binom.pmf(ks, 1000, p)
    plt.plot(ks, pk, 'r', label='Binomial distribution')
    plt.legend()
    plt.show(True)

def generate_erdos_renyi(N, avg_degree, dataset):
    p = avg_degree/(N-1)
    GER = nx.gnp_random_graph(N, p, seed=None, directed=False)
    dataset.append(GER)

def generate_watts_strogatz(N, avg_degree, p, dataset):
    k = int(avg_degree/2)
    GWS = nx.watts_strogatz_graph(N, k, p, seed=None)
    dataset.append(GWS)

def generate_barabasi_albert(N, avg_degree, dataset):
    m = int(avg_degree/2)
    GBA = nx.barabasi_albert_graph(N, m)
    dataset.append(GBA)

def process_graph(G, n_networks, mean_matrix, deviation_matrix, line):
    n_nodes_list = list()
    avg_degree_list = list()
    smd_distribution_list = list()
    avg_shortest_list = list()
    avg_clustering_list = list()
    transitivity_list = list()
    assortativity_list = list()
    for i in range(0, n_networks):
        N = len(G[i])
        M = G[i].number_of_edges()
        n_nodes_list.append(N)
        avg_degree_list.append(2*M/N)
        smd_distribution_list.append(second_momment_of_degree_distribution(G[i]))
        avg_shortest_list.append(nx.average_shortest_path_length(G[i]))
        avg_clustering_list.append(nx.average_clustering(G[i]))
        transitivity_list.append(nx.transitivity(G[i]))
        assortativity_list.append(nx.degree_assortativity_coefficient(G[i]))
    a = np.array(n_nodes_list)
    mean_matrix[line][0] = np.mean(a)
    deviation_matrix[line][0] = np.std(a)

    a = np.array(avg_degree_list)
    mean_matrix[line][1] = np.mean(a)
    deviation_matrix[line][1] = np.std(a)

    a = np.array(smd_distribution_list)
    mean_matrix[line][2] = np.mean(a)
    deviation_matrix[line][2] = np.std(a)

    a = np.array(avg_shortest_list)
    mean_matrix[line][3] = np.mean(a)
    deviation_matrix[line][3] = np.std(a)

    a = np.array(avg_clustering_list)
    mean_matrix[line][4] = np.mean(a)
    deviation_matrix[line][4] = np.std(a)

    a = np.array(transitivity_list)
    mean_matrix[line][5] = np.mean(a)
    deviation_matrix[line][5] = np.std(a)

    a = np.array(assortativity_list)
    mean_matrix[line][6] = np.mean(a)
    deviation_matrix[line][6] = np.std(a)

For some reason the distributions are looking weird, but it's possible to draw some conclusions from them. We can se that the in the Erds-Ryni networks the degree is very centered in the average degree that we set previously. The Barabsi-Albert has a lot of nodes with a low degree due to its scale-free nature. The Watts-Strogatz network with p = 0.1 has a more diverse range of degree than the one with p = 0.01, however both have degress centered around 4. The Watts-Strogatz have the higher clustering coefficient, transitivity and average shortest path length. Barabsi-Albert has the highest sendom momment of degree distribution.¶
